\section{A statistical test for RFI's}\label{sec:main}
In this section we present our algorithm to mine a collection of RFI's
while guaranteeing that the FWER is within the user-specified parameter
$\delta$. We use results from statistical learning theory to achieve this goal.
Specifically, we define a range space where points are transactions, and ranges
are specific sets of transactions associated to the problem at hand. We then
show how to bound the VC-Dimension of this range space and that this bound is
strict. In Section~\ref{sec:algo}, we present our algorithm to mine a
collection of RFI's.

\subsection{Algorithm}\label{sec:algo}
\begin{algorithm}[ht]
  \SetKwInOut{Input}{Input}
  \SetKwInOut{Output}{Output}
  \SetKwFunction{VCIndexUpperBound}{VCIndexUpperBound}
  \SetKwFunction{GetEpsilonVC}{GetEpsilonVC}
  \SetKwFunction{GetEpsilonEmpiricalVC}{GetEpsilonEmpiricalVC}
  \SetKwFunction{GetDIndex}{GetDIndex}
  \DontPrintSemicolon
  %\dontprintsemicolon
  \Input{a dataset $\Ds$, a minimum probability mass threshold $\theta$, a FWER
    bound $\delta$.}
  \Output{a collection of itemsets $A$ with $p(A)\ge\theta$ such that FWER $<\delta$.}
  %$d_1\leftarrow\left\lfloor\log_2\left(\binom{m}{\lfloor m/2\rfloor}+1\right)\right\rfloor$\;
  $\delta_1,\delta_2\leftarrow 1-\sqrt{1-\delta}$\; 
  $\varepsilon_{1,\mbox{VC}}\leftarrow$ \GetEpsilonVC{$|\Itm|-1, |\Ds|, \delta_1$}\;
  $\mbox{d-Index}\leftarrow$ \GetDIndex{$\Ds$}\;
  $\varepsilon_{1,\mbox{empVC}}\leftarrow$
  \GetEpsilonEmpiricalVC{$\mbox{d-Index}, |\Ds|, \delta_1$}\;
  $\varepsilon_1\leftarrow\min\{\varepsilon_{1,\mbox{VC}},\varepsilon_{1,\mbox{empVC}}}$\;
  $\mbox{lowerBoundFIs}\leftarrow\FI(\Ds,\Itm,\theta-\varepsilon_1/2)$\;
  $\mbox{lowerBoundNegativeBorder}\leftarrow$ \GetNegativeBorder{$A$}\;
  $\mbox{toRemove}\leftarrow\FI(\Ds,\Itm,\theta+\varepsilon_1/2)$\;
  $\mbox{negativeBorderCandidates}\leftarrow (\mbox{lowerBoundFIs} \setminus
  \mbox{toRemove}) \cup \mbox{lowerBoundNegativeBorder}$\;
  $\mbox{VCDimUpperBound}\leftarrow$
  \GetVCDimUpperBound{$\Itm,\mbox{negativeBorderCandidates}$}\;
  $\mbox{empVCDimUpperBound} \leftarrow$ \GetEmpVCDimUpperBound{$\Ds,\Itm,\mbox{negativeBorderCandidates}$}\;
  $\varepsilon_{2,\mbox{VC}} \leftarrow$ \GetEpsilonVC{$\mbox{VCDimUpperBound}, |\Ds|, \delta_2$}\;
  $\varepsilon_{2,\mbox{empVC}} \leftarrow$ \GetEpsilonEmpiricalVC{$\mbox{empVCDimUpperBound}, |\Ds|, \delta_1$};
  $\varepsilon_2\leftarrow\min\{\varepsilon_{2,\mbox{VC}},\varepsilon_{2,\mbox{empVC}}\}$\;
  \Return{$\FI\left(\Ds,\Itm,\theta+\varepsilon_2/2\right)$}\;
  \caption{A statistical test for RFI's}\label{alg:pseudocode}
\end{algorithm}
